Sub ExporteerGrafiekAlsPNG()
    Dim pad As String
    Dim bestandsnaam As String
    
    pad = "C:\Users\stijn\OneDrive - Proces Bouwers\1.Projecten\Claude\Omgevingsscan\Huidige scan\"
    bestandsnaam = "Doorlooptijd"
    
    ' Map aanmaken als die nog niet bestaat
    If Dir(pad, vbDirectory) = "" Then
        MkDir pad
    End If
    
    ' Grafiek exporteren als PNG
    ActiveSheet.ChartObjects("Grafiek 1").Chart.Export _
        Filename:=pad & bestandsnaam & ".png", _
        FilterName:="PNG"
    
    MsgBox "Opgeslagen: " & pad & bestandsnaam & ".png", vbInformation
End Sub
